\NewDocumentCommand{\ctest}{}
  {
    \clist_if_empty:NTF \l_test_tmpa_clist
      {
        empty\par
      }
      {
        not~empty\par
        \clist_map_inline:Nn \l_test_tmpa_clist
          {
            $<$##1$>$~
          }
          \par
      }
  }
  
\prg_new_protected_conditional:Npnn \ctestA:nn #1#2 {T, F, TF}  
  {
    \str_if_eq:nnTF {#1} {#2}
      {
        they~are~equal\par
        \prg_return_true:
      }
      {
        they~are~not~equal\par
        \prg_return_false:
      }
  }
  

\cs_new_protected:Npn \ctestB:nnTF #1#2
  {
    \ctestA:nnTF{#1}{#2}
  }  
  
\NewDocumentCommand{\testA}{mmmm}  
  {
    \ctestA:nnTF {#1}{#2}{#3}{#4}
  }

  {
    \ctestB:nnTF {#1}{#2}{#3}{#4}
  }
